13def checkerboard( 

14 width: int, height: int, count_x: int = 8, count_y: int = 8 

15) -> np.ndarray: 

16 """Generate a black-and-white checkerboard test image. 

17 

18 Same parameters and pixel values as the `dictk checkerboard` CLI 

19 command, minus the file write — see the `dictk` package docstring for 

20 why the API layer stops at the array. count_x and count_y are named for 

21 the rectangles they divide the image into, not "squares": unequal 

22 counts (or a width/height ratio that doesn't match count_x/count_y) 

23 produce rectangular cells, not square ones. 

24 

25 Args: 

26 width: Image width in pixels. 

27 height: Image height in pixels. 

28 count_x: Number of rectangles along the width. Must be >= 1. 

29 count_y: Number of rectangles along the height. Must be >= 1. 

30 

31 Returns: 

32 A 2D uint8 array of shape (height, width) with values 0 or 255. 

33 

34 Raises: 

35 ValueError: If count_x or count_y is less than 1. 

36 """ 

37 if count_x < 1: 

38 raise ValueError(f"count_x {count_x} must be >= 1") 

39 if count_y < 1: 

40 raise ValueError(f"count_y {count_y} must be >= 1") 

41 

42 rows = (np.arange(height) * count_y // height) % 2 

43 cols = (np.arange(width) * count_x // width) % 2 

44 pattern = np.logical_xor(rows[:, None], cols[None, :]) 

45 return (pattern * 255).astype(np.uint8)

48def astronaut(width: int = 512, height: int = 512) -> np.ndarray: 

49 """Load a bundled real-world grayscale reference image. 

50 

51 Same parameters and pixel values as the `dictk astronaut` CLI 

52 command, minus the file write — see the `dictk` package docstring for 

53 why the API layer stops at the array. 

54 

55 The source is a NASA portrait of astronaut Eileen Collins, from the 

56 NASA Great Images database ("No known copyright restrictions, released 

57 into the public domain."). It's bundled as a color image and converted 

58 with `rgba_to_gray`; unlike `rosta` and `checkerboard`, it isn't 

59 procedurally generated, so `width`/`height` other than the native 

60 512x512 resize the source image (via `scipy.ndimage.zoom`) rather than 

61 computing a fresh pattern at that size. 

62 

63 Args: 

64 width: Image width in pixels. Must be >= 1. 

65 height: Image height in pixels. Must be >= 1. 

66 

67 Returns: 

68 A 2D uint8 array of shape (height, width). 

69 

70 Raises: 

71 ValueError: If width or height is less than 1. 

72 """ 

73 if width < 1: 

74 raise ValueError(f"width {width} must be >= 1") 

75 if height < 1: 

76 raise ValueError(f"height {height} must be >= 1") 

77 

78 asset_path = importlib.resources.files("dictk") / "data" / "astronaut.png" 

79 with importlib.resources.as_file(asset_path) as path: 

80 color = read_image(path) 

81 gray = rgba_to_gray(color) 

82 

83 native_height, native_width = gray.shape 

84 if (width, height) == (native_width, native_height): 

85 return gray 

86 

87 zoom_factors = (height / native_height, width / native_width) 

88 resized = zoom(gray.astype(np.float64), zoom_factors, order=3) 

89 return np.clip(resized, 0, 255).astype(np.uint8)

104def rgba_to_gray(arr: np.ndarray) -> np.ndarray: 

105 """Convert an RGB(A) image to grayscale by averaging the RGB channels. 

106 

107 Args: 

108 arr: Input image array, either 2D (grayscale) or 3D (color). 

109 

110 Returns: 

111 A 2D grayscale image array. If the input is already 2D, it is 

112 returned unchanged. 

113 

114 Raises: 

115 ValueError: If the array is neither 2D nor a 3-or-4-channel 3D array. 

116 """ 

117 if arr.ndim == 2: 

118 return arr 

119 

120 if is_rgba(arr): 

121 return np.mean(arr[:, :, :3], axis=2).astype(arr.dtype) 

122 

123 raise ValueError( 

124 "Input array must be either 2D (grayscale) or 3D with 3 or 4 channels (color)." 

125 )

128def combine_images(a: np.ndarray, b: np.ndarray) -> np.ndarray: 

129 """Combine two images by averaging their pixel values. 

130 

131 Args: 

132 a: First image, 2D grayscale or 3D color. 

133 b: Second image, same shape as `a` once converted to grayscale. 

134 

135 Returns: 

136 A 2D uint8 array, normalized to the range [0, 255]. 

137 

138 Raises: 

139 ValueError: If the grayscale-converted images don't share a shape. 

140 """ 

141 gray_a = rgba_to_gray(a) 

142 gray_b = rgba_to_gray(b) 

143 

144 if gray_a.shape != gray_b.shape: 

145 raise ValueError( 

146 f"shape mismatch: a.shape={gray_a.shape}, b.shape={gray_b.shape}" 

147 ) 

148 

149 combined = gray_a.astype(np.float64) + gray_b.astype(np.float64) 

150 return (combined / combined.max() * 255).astype(np.uint8)

153def brightness(arr: np.ndarray, factor: float) -> np.ndarray: 

154 """Adjust image brightness by an additive shift, clipped to [0, 255]. 

155 

156 Brightness *translates* the pixel-intensity histogram: every pixel is 

157 shifted by the same amount, so dark areas lighten right along with 

158 bright ones (unlike a multiplicative scale, where a black pixel would 

159 stay exactly black). factor=1.0 leaves the image unchanged; factor=1.5 

160 shifts every pixel by +127.5*0.5 = +63.75, factor=2.0 by +127.5. 

161 

162 Args: 

163 arr: A 2D grayscale image array. 

164 factor: Brightness factor. 1.0 is unchanged; > 1.0 brightens 

165 (shifts toward white); < 1.0 darkens (shifts toward black). 

166 

167 Returns: 

168 A 2D uint8 array, same shape as `arr`. 

169 """ 

170 max_pixel_value = 255.0 

171 offset = (factor - 1.0) * (max_pixel_value / 2.0) 

172 shifted = arr.astype(np.float64) + offset 

173 return np.clip(shifted, 0, max_pixel_value).astype(np.uint8)

176def contrast(arr: np.ndarray, factor: float) -> np.ndarray: 

177 """Adjust image contrast by scaling around the mean, clipped to [0, 255]. 

178 

179 Contrast *stretches* the pixel-intensity histogram outward from its own 

180 mean, rather than shifting it: the mean stays roughly the same, while 

181 values spread further from it. factor=1.0 leaves the image unchanged; 

182 factor=0.0 collapses every pixel to the mean (a flat gray image); 

183 factor > 1.0 pushes values further toward 0 and 255. 

184 

185 Args: 

186 arr: A 2D grayscale image array. 

187 factor: Contrast factor. 1.0 is unchanged; > 1.0 increases 

188 contrast; between 0.0 and 1.0 decreases it. 

189 

190 Returns: 

191 A 2D uint8 array, same shape as `arr`. 

192 """ 

193 mean = arr.astype(np.float64).mean() 

194 stretched = (arr.astype(np.float64) - mean) * factor + mean 

195 return np.clip(stretched, 0, 255).astype(np.uint8)

210def write_svg(arr: np.ndarray, path: Path) -> None: 

211 """Write a NumPy array to an SVG file. 

212 

213 SVG is a vector format with no native pixel-grid concept, so the array 

214 is PNG-encoded and embedded as a base64 data URI inside a minimal SVG 

215 wrapper (the standard way to carry raster data in SVG) rather than 

216 traced into vector shapes. 

217 

218 Args: 

219 arr: The image array to save. 

220 path: The output file path. 

221 """ 

222 height, width = arr.shape[:2] 

223 png_bytes = iio.imwrite("<bytes>", arr, extension=".png") 

224 encoded = base64.b64encode(png_bytes).decode("ascii") 

225 

226 svg = ( 

227 '<?xml version="1.0" encoding="UTF-8"?>\n' 

228 f'<svg xmlns="http://www.w3.org/2000/svg" ' 

229 f'xmlns:xlink="http://www.w3.org/1999/xlink" ' 

230 f'width="{width}" height="{height}" viewBox="0 0 {width} {height}">\n' 

231 f' <image width="{width}" height="{height}" ' 

232 f'xlink:href="data:image/png;base64,{encoded}"/>\n' 

233 "</svg>\n" 

234 ) 

235 Path(path).write_text(svg, encoding="ascii")

238def write_image(arr: np.ndarray, path: Path) -> None: 

239 """Write a NumPy array to an image file. 

240 

241 Dispatches on the file extension: `.svg` is handled by `write_svg` 

242 (embedding a raster PNG in an SVG wrapper); every other extension 

243 (`.tiff`, `.png`, `.jpg`, ...) is handled by imageio directly. 

244 

245 Args: 

246 arr: The image array to save. 

247 path: The output file path. 

248 """ 

249 if Path(path).suffix.lower() == ".svg": 

250 write_svg(arr, path) 

251 return 

252 

253 iio.imwrite(path, arr)

256def describe_image(arr: np.ndarray) -> str: 

257 """Format a description of an image array's type, shape, and color format. 

258 

259 Args: 

260 arr: The image array to describe. 

261 

262 Returns: 

263 A multi-line description string. 

264 """ 

265 lines = [ 

266 f"Type: {type(arr)}", 

267 f"Shape: {arr.shape}", 

268 f"Dtype: {arr.dtype}", 

269 ] 

270 

271 match arr.shape: 

272 case (_height, _width): 

273 lines.append("The image is grayscale.") 

274 case (_height, _width, 3): 

275 lines.append("The image is color (RGB).") 

276 case (_height, _width, 4): 

277 lines.append("The image is color (RGBA, includes alpha channel).") 

278 case _: 

279 lines.append("The image has an unsupported format.") 

280 

281 return "\n".join(lines)

284def save_histogram(arr: np.ndarray, path: Path, dpi: int = 300) -> None: 

285 """Save a histogram of pixel intensities [0, 255] for a grayscale image. 

286 

287 Args: 

288 arr: The 2D grayscale image array, expected type uint8, range [0, 255]. 

289 path: The output file path for the histogram image. 

290 dpi: Resolution of the saved figure. 

291 """ 

292 plt.figure() 

293 plt.hist(arr.ravel(), bins=256, range=(0, 255), color="black", alpha=0.7) 

294 plt.title(f"pixel histogram intensity\n{path}", fontsize=8)
